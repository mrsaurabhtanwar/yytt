Diagram 1: Grid with Shapes

\begin{pspicture}(8,8)
  \psgrid(0,0)(8,8)
  % dotted red segment
  \psline[linestyle=dotted,linecolor=red,linewidth=2pt](2,3)(4,5)
  % blue triangle
  \psline[linecolor=blue](1,1)(5,1)(1,4)(1,1)
  % hollow circle
  \pscircle[linestyle=dotted](3,2.5){2.5}
  % filled small circle
  \pscircle[fillstyle=solid,fillcolor=blue](2,2){1}
  % vectors
  \psline{<-}(5,5)(7,7)
  \psline{<->}(1.5,5.5)(3.5,5.5)
  % filled dot and cross dots
  \pscircle*(6,6){1}
  \psdots[dotstyle=+](1.5,6.5)(1.5,7.5)
  \psdots[dotstyle=o](1.8,6.8)
\end{pspicture}

Diagram 6: Clipped Ellipse

  \psgrid(0,0)(9,9)

\end{pspicture}

Diagram 8: Cross Pattern

\begin{pspicture}(5,4)
  \psset{unit=0.5}
  \psgrid(0,0)(13,13)
  \def\cross{%
    \psline[fillstyle=solid,fillcolor=green]
      (0,0)(1,0)(1,1)(2,1)(2,2)(1,2)(1,3)(0,3)(0,2)(-1,2)(-1,1)(0,1)(0,0)%
  }
  \put(3,1){\cross}
  \put(6,2){\cross}
  \put(9,3){\cross}
  \rput{60}(10,10){\cross}
\end{pspicture}

Diagram 9: Sine Wave Function

\begin{pspicture}(-3,-4.25)(7.5,4.25)
  \psset{xunit=2cm,yunit=2cm}
  \psaxes[linecolor=red,linewidth=0.1pt]{<->}(5,3)
  \psplot[algebraic=true,plotpoints=2500,linecolor=blue]
    {0.025}{2}{sin(1/x)}
  \put(7,-0.5){$x$}
  \put(-0.5,5){$y$}
  \put(3,3){$y=\sin\dfrac1x$}
\end{pspicture}

Diagram 10: Square Root and Quadratic Function

\begin{pspicture}(-0.5,-0.5)(5,5.5)
  \psset{xunit=4cm,yunit=4cm,plotpoints=500}
  \psaxes{->}(0,0)(1.25,1.25)
  \psplot[algebraic=true,linecolor=red]{0}{1}{sqrt(x)}
  \psplot[algebraic=true,linecolor=blue]{0}{1}{x^2}
  \put(5,-0.5){$x$}
  \put(-0.5,5){$y$}
  \put(1.75,3.75){$y=\sqrt{x}$}
\end{pspicture}
Image 11

Diagram 11: Parametric Plot with Arrows

\begin{pspicture}(-2.5,-2.5)(2.5,2.5)
  \psset{xunit=3cm,yunit=3cm}
  \psaxes[ticks=none,labels=none]{<->}(0,0)(-2.5,-0.75)(2.5,0.75)
  \parametricplot[plotpoints=500,arrows=->,arrowscale=1.5]
    {0}{360}{1 t cos add t sin}
  \put(4.5,-0.5){$x$}
  \put(-0.5,2.5){$y$}
\end{pspicture}
